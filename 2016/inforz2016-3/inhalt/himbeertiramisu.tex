\artikelohnevorspann{Himbeertiramisu}
{\paragraph{Zutaten:}
\begin{itemize}
    \item 500\,g Himbeeren (TK)
    \item 500\,g Magerquark
    \item 2 Becher Schlagsahne
    \item 1-2 Päckchen Sahnesteif (opt.)
    \item 100\,g Zucker
    \item 200-300\,g Löffelbiscuits
    \item Zimt
\end{itemize}
\vspace*{\fill}
\columnbreak
\begin{enumerate}
    \item Himbeeren antauen lassen und pürieren.
    \item Sahne steif schlagen. Magerquark und Zucker mischen, steife Sahne
        unterrühren.
    \item In eine Form / Schüssel eine Schicht Sahnequark streichen. Löffelbiscuits
        darauf verteilen. Darauf kommt eine Schicht pürierte Himbeeren. Nun weiter
        Sahnequark, Biscuits und Himbeeren schichten. Mit Sahnequark abschließen.
    \item Zum Schluss mit Zimt bestreuen und mehrere Stunden im Kühlschrank durchziehen
        lassen.
\end{enumerate}
}
{Tobias Otterbein}
